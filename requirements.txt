# -------------------------
# Quick install / run commands
# -------------------------
# Install system deps (Ubuntu/Debian) - example:
# sudo apt update && sudo apt install -y ca-certificates curl gnupg lsb-release \
#   build-essential pkg-config python3 python3-dev libvips-dev libvips-tools

# Install Docker (see README or official Docker docs). Example commands are in
# the repository README.md under 'Docker (docker-compose)'.

# Install Node and npm (nvm recommended):
# curl -fsSL https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.3/install.sh | bash
# source ~/.nvm/nvm.sh
# nvm install 18 && nvm use 18

# Install project dependencies:
# cd backend && npm install
# cd ../frontend && npm install

# Run locally (dev):
# cd backend && npm run dev
# cd frontend && npm run dev

# Or run the Docker stack (recommended for production):
# docker compose up -d --build
